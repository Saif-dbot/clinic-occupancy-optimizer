\begin{frame}{Objectif}
  \begin{itemize}
    \item Presenter clairement comment l application fonctionne aujourd hui.
    \item Expliquer la structure technique frontend et backend.
    \item Donner un plan de migration page par page vers l API FastAPI.
  \end{itemize}
\end{frame}

\begin{frame}{Vision rapide du projet}
  MediPlan est une application de gestion des rendez-vous medicaux orientee reduction du no-show.
  \vspace{0.4cm}
  \begin{itemize}
    \item Frontend Next.js (UI et navigation)
    \item Backend FastAPI (logique metier et endpoints OFS)
    \item Donnees mock cote frontend pour la demo actuelle
  \end{itemize}
\end{frame}

\begin{frame}{Probleme metier adresse}
  \begin{itemize}
    \item Absences patients (no-show) qui degradent le taux d occupation.
    \item Creneaux annules non reaffectes a temps.
    \item Charge forte pour secretariat et pilotage difficile des KPI.
  \end{itemize}
  \vspace{0.3cm}
  \textbf{Cible:} ameliorer occupation, reaffectation et qualite operationnelle.
\end{frame}

\begin{frame}{Architecture actuelle}
  \textbf{Frontend}
  \begin{itemize}
    \item Point d entree: \texttt{app/layout.tsx}
    \item Login: \texttt{app/page.tsx}
    \item Routes protegees: \texttt{app/(app)/layout.tsx}
    \item Shell navigation: \texttt{components/layout/app-shell.tsx}
    \item Donnees demo: \texttt{lib/mock-api.ts}, \texttt{lib/mock-data.ts}
  \end{itemize}
  \vspace{0.2cm}
  \textbf{Backend}
  \begin{itemize}
    \item API: \texttt{backend/app/main.py}
  \end{itemize}
\end{frame}

\begin{frame}{Flux fonctionnel de bout en bout}
  \begin{enumerate}
    \item Utilisateur sur la page de connexion.
    \item Validation du compte demo via contexte applicatif.
    \item Redirection vers \texttt{/dashboard} apres authentification.
    \item Verification d acces via layout protege.
    \item Chargement des donnees via \texttt{mock-api.ts} (etat actuel).
    \item Navigation et permissions selon le role actif.
  \end{enumerate}
\end{frame}

\begin{frame}{Endpoints OFS deja disponibles}
  \begin{itemize}
    \item \texttt{POST /appointments}
    \item \texttt{PATCH /appointments/\{id\}/cancel}
    \item \texttt{POST /waitlist/\{siteId\}/enqueue}
    \item \texttt{GET /dashboard/kpi}
    \item \texttt{POST /risk-score/recompute/\{appointmentId\}}
  \end{itemize}
  \vspace{0.3cm}
  \textbf{Objectif de la prochaine phase:} brancher progressivement les ecrans sur ces endpoints.
\end{frame}

\begin{frame}{Plan de migration - Etape A (fondations)}
  \begin{enumerate}
    \item Creer un client API unique (\texttt{lib/api-client.ts}).
    \item Ajouter \texttt{NEXT\_PUBLIC\_API\_BASE\_URL}.
    \item Definir un fallback mock en dev si backend indisponible.
  \end{enumerate}
\end{frame}

\begin{frame}{Plan page par page - Etape B (2/2)}

\begin{frame}{Criteres de validation pour chaque page}
